const addOffer = () => {
  // Validate required fields
  if (!newOffer.offerName) {
    alert("Please enter Offer Name");
    return;
  }
  if (!newOffer.discount) {
    alert("Please enter Discount percentage");
    return;
  }
  
  // Create offer object with all fields
  const offer = {
    id: offers.length + 1,
    offerName: newOffer.offerName,
    productName: newOffer.productName || "All Products",
    bulkQuantity: newOffer.bulkQuantity || "N/A",
    bulkValue: newOffer.bulkValue || "N/A",
    discount: parseFloat(newOffer.discount) || 0,
    freeDelivery: newOffer.freeDelivery || "No",
    buyGet: newOffer.buyGet || "N/A",
    billDiscount: newOffer.billDiscount || "N/A",
    validFrom: newOffer.validFrom || new Date().toISOString().split("T")[0],
    validTo: newOffer.validTo || new Date(Date.now() + 30*24*60*60*1000).toISOString().split("T")[0],
    status: "Active",
    ordersReceived: 0
  };
  
  setOffers([...offers, offer]);
  setSupplierInfo({ ...supplierInfo, activeOffers: supplierInfo.activeOffers + 1 });
  
  // Reset form
  setNewOffer({ 
    offerName: "", 
    productName: "", 
    bulkQuantity: "", 
    bulkValue: "", 
    discount: "", 
    freeDelivery: "No",
    buyGet: "",
    billDiscount: "",
    validFrom: "", 
    validTo: "" 
  });
  setShowOfferModal(false);
  alert("✅ Offer created successfully!");
};
