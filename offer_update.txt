// Replace the addOffer function in supplier-dashboard.tsx
const addOffer = () => {
  if (!newOffer.offerName || !newOffer.discount) {
    alert("Please fill offer name and discount");
    return;
  }
  
  const offer = {
    id: offers.length + 1,
    offerName: newOffer.offerName,
    productName: newOffer.productName || "All Products",
    bulkQuantity: newOffer.bulkQuantity || "N/A",
    bulkValue: newOffer.bulkValue || "N/A",
    discount: parseFloat(newOffer.discount),
    freeDelivery: newOffer.freeDelivery || "No",
    validFrom: newOffer.validFrom,
    validTo: newOffer.validTo,
    status: "Active",
    ordersReceived: 0
  };
  
  setOffers([...offers, offer]);
  setSupplierInfo({ ...supplierInfo, activeOffers: supplierInfo.activeOffers + 1 });
  setNewOffer({ 
    offerName: "", 
    productName: "", 
    bulkQuantity: "", 
    bulkValue: "", 
    discount: "", 
    freeDelivery: "No", 
    validFrom: "", 
    validTo: "" 
  });
  setShowOfferModal(false);
  alert("Offer created successfully!");
};

// Update the Offer Modal JSX
showOfferModal && React.createElement("div", { style: styles.modal, onClick: () => setShowOfferModal(false) },
  React.createElement("div", { style: styles.modalContent, onClick: (e) => e.stopPropagation() },

    React.createElement("div", null,
      React.createElement("label", { style: styles.label }, "Offer Name *"),

    ),

    ),
